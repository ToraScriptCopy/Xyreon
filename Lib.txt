local function round(num, bracket)
	bracket = bracket or 1
	return math.floor(num/bracket + 0.5) * bracket
end

local function keyCheck(x,x1)
	for _,v in next, x1 do
		if v == x then
			return true
		end
	end
end

local function update(input)

end

function library:Create(class, properties)
	properties = typeof(properties) == "table" and properties or {}
	local inst = Instance.new(class)
	for property, value in next, properties do
		inst[property] = value
	end
	return inst
end

--[[ NEW: SHA-256 Hashing for Key System ]]
local function sha256(str)
	local function rightrotate(x, n)
		return (x - (x % (2^n))) / (2^n) + (x % (2^n)) * (2^(32-n))
	end
	local function to_binary_string(num)
		local result = ""
		for i = 31, 0, -1 do
			result = result .. (math.floor(num / (2^i)) % 2)
		end
		return result
	end
	local function from_binary_string(bin)
		local num = 0
		for i = 1, #bin do
			num = num + (tonumber(string.sub(bin, i, i)) * (2^(#bin-i)))
		end
		return num
	end
	local h = {0x6a09e667, 0xbb67ae85, 0x3c6ef372, 0xa54ff53a, 0x510e527f, 0x9b05688c, 0x1f83d9ab, 0x5be0cd19}
	local k = {0x428a2f98, 0x71374491, 0xb5c0fbcf, 0xe9b5dba5, 0x3956c25b, 0x59f111f1, 0x923f82a4, 0xab1c5ed5, 0xd807aa98, 0x12835b01, 0x243185be, 0x550c7dc3, 0x72be5d74, 0x80deb1fe, 0x9bdc06a7, 0xc19bf174, 0xe49b69c1, 0xefbe4786, 0x0fc19dc6, 0x240ca1cc, 0x2de92c6f, 0x4a7484aa, 0x5cb0a9dc, 0x76f988da, 0x983e5152, 0xa831c66d, 0xb00327c8, 0xbf597fc7, 0xc6e00bf3, 0xd5a79147, 0x06ca6351, 0x14292967, 0x27b70a85, 0x2e1b2138, 0x4d2c6dfc, 0x53380d13, 0x650a7354, 0x766a0abb, 0x81c2c92e, 0x92722c85, 0xa2bfe8a1, 0xa81a664b, 0xc24b8b70, 0xc76c51a3, 0xd192e819, 0xd6990624, 0xf40e3585, 0x106aa070, 0x19a4c116, 0x1e376c08, 0x2748774c, 0x34b0bcb5, 0x391c0cb3, 0x4ed8aa4a, 0x5b9cca4f, 0x682e6ff3, 0x748f82ee, 0x78a5636f, 0x84c87814, 0x8cc70208, 0x90befffa, 0xa4506ceb, 0xbef9a3f7, 0xc67178f2}
	local message = ""
	for i = 1, #str do
		message = message .. string.format("%08d", string.format("%d", string.byte(str, i)))
	end
	local l = #message
	message = message .. "1"
	while #message % 512 ~= 448 do
		message = message .. "0"
	end
	local l_binary = to_binary_string(l)
	while #l_binary < 64 do
		l_binary = "0" .. l_binary
	end
	message = message .. l_binary
	for i = 1, #message / 512 do
		local chunk = string.sub(message, (i-1)*512 + 1, i*512)
		local w = {}
		for j = 1, 16 do
			w[j] = from_binary_string(string.sub(chunk, (j-1)*32 + 1, j*32))
		end
		for j = 17, 64 do
			local s0 = rightrotate(w[j-15], 7) ~ rightrotate(w[j-15], 18) ~ (w[j-15] - (w[j-15] % 8)) / 8
			local s1 = rightrotate(w[j-2], 17) ~ rightrotate(w[j-2], 19) ~ (w[j-2] - (w[j-2] % 1024)) / 1024
			w[j] = (w[j-16] + s0 + w[j-7] + s1) % (2^32)
		end
		local a, b, c, d, e, f, g, h_ = h[1], h[2], h[3], h[4], h[5], h[6], h[7], h[8]
		for j = 1, 64 do
			local s1 = rightrotate(e, 6) ~ rightrotate(e, 11) ~ rightrotate(e, 25)
			local ch = (e & f) ~ ((~e) & g)
			local temp1 = (h_ + s1 + ch + k[j] + w[j]) % (2^32)
			local s0 = rightrotate(a, 2) ~ rightrotate(a, 13) ~ rightrotate(a, 22)
			local maj = (a & b) ~ (a & c) ~ (b & c)
			local temp2 = (s0 + maj) % (2^32)
			h_ = g
			g = f
			f = e
			e = (d + temp1) % (2^32)
			d = c
			c = b
			b = a
			a = (temp1 + temp2) % (2^32)
		end
		h[1] = (h[1] + a) % (2^32)
		h[2] = (h[2] + b) % (2^32)
		h[3] = (h[3] + c) % (2^32)
		h[4] = (h[4] + d) % (2^32)
		h[5] = (h[5] + e) % (2^32)
		h[6] = (h[6] + f) % (2^32)
		h[7] = (h[7] + g) % (2^32)
		h[8] = (h[8] + h_) % (2^32)
	end
	local digest = ""
	for i = 1, 8 do
		digest = digest .. string.format("%08x", h[i])
	end
	return digest
end

--[[ UPGRADED: Secure Key System Dialog ]]
function library:CreateKeyDialog(options)
	options = options or {}
	local dialog = {}
	
	local title = options.Title or "Authentication Required"
	local successCallback = options.OnSuccess or function() print("Key Accepted") end
	--[[
		SECURITY NOTE:
		Client-side checks are ALWAYS bypassable. For real security, this function
		should use a RemoteFunction to ask the server if the key is valid.
		
		Example with SHA-256 (more secure client-side):
		local HASHED_KEY = "03ac674216f3e15c761ee1a5e255f067953623c8b388b4459e13f978d7c846f4" -- This is sha256("1234")
		CheckKey = function(key) return sha256(key) == HASHED_KEY end
	]]
	local checkCallback = options.CheckKey or function(key) return key == "1234" end
	local failureCallback = options.OnFailure or function()
		library.base:Destroy()
		game:GetService("Players").LocalPlayer:Kick("Authentication Failed.")
	end

	local overlay = self:Create("Frame", {
		Name = "KeyDialogOverlay",
		Size = UDim2.new(1, 0, 1, 0),
		BackgroundColor3 = Color3.new(0, 0, 0),
		BackgroundTransparency = 1,
		ZIndex = 100,
		Visible = false,
		Parent = self.base
	})

	local main = self:Create("Frame", {
		Name = "KeyDialog",
		AnchorPoint = Vector2.new(0.5, 0.5),
		Position = UDim2.new(0.5, 0, 0.45, 0),
		Size = UDim2.new(0, 320, 0, 180),
		BackgroundColor3 = self.theme.Background,
		BackgroundTransparency = 0.1,
		BorderSizePixel = 0,
		Parent = overlay
	})
	self:Create("UICorner", { CornerRadius = UDim.new(0, self.theme.CornerRounding), Parent = main })
	self:Create("UIStroke", { Color = self.theme.Outline, Thickness = 1, Parent = main })
	self:Create("UIAspectRatioConstraint", { AspectRatio = 1.77, Parent = main })

	local titleLabel = self:Create("TextLabel", {
		Name = "Title",
		Size = UDim2.new(1, 0, 0, 48),
		BackgroundColor3 = self.theme.Surface,
		BackgroundTransparency = 0.5,
		Font = self.theme.Font.Bold,
		Text = "  " .. title,
		TextColor3 = self.theme.OnBackground,
		TextSize = 18,
		TextXAlignment = Enum.TextXAlignment.Left,
		Parent = main
	})
	self:Create("UICorner", { CornerRadius = UDim.new(0, self.theme.CornerRounding), Parent = titleLabel })
	local titleIcon = self:Create("ImageLabel", {
		Size = UDim2.new(0, 24, 0, 24),
		Position = UDim2.new(0, 12, 0.5, 0),
		AnchorPoint = Vector2.new(0, 0.5),
		BackgroundTransparency = 1,
		Image = self.icons['key-round'] or "",
		ImageColor3 = self.theme.Muted,
		Parent = titleLabel
	})
	titleLabel.Text = "      " .. title

	local statusLabel = self:Create("TextLabel", {
		Name = "Status",
		Position = UDim2.new(0, 0, 0, 145),
		Size = UDim2.new(1, 0, 0, 20),
		BackgroundTransparency = 1,
		Font = self.theme.Font.Regular,
		Text = "",
		TextColor3 = self.theme.Muted,
		TextSize = 14,
		Parent = main
	})

	local keyBox = self:Create("TextBox", {
		Name = "KeyInput",
		Position = UDim2.new(0.5, -125, 0.5, -10),
		Size = UDim2.new(0, 250, 0, 40),
		BackgroundColor3 = self.theme.Surface,
		Font = self.theme.Font.Regular,
		PlaceholderText = "Enter your key...",
		PlaceholderColor3 = self.theme.Muted,
		Text = "",
		TextColor3 = self.theme.OnSurface,
		TextSize = 16,
		ClearTextOnFocus = false,
		Parent = main
	})
	self:Create("UICorner", { CornerRadius = UDim.new(0, self.theme.ElementRounding), Parent = keyBox })
	local keyBoxStroke = self:Create("UIStroke", { Color = self.theme.Outline, Thickness = 1, Parent = keyBox })
	
	keyBox.Focused:Connect(function() tweenService:Create(keyBoxStroke, TweenInfo.new(0.2), {Color = self.theme.Primary}):Play() end)
	keyBox.FocusLost:Connect(function() tweenService:Create(keyBoxStroke, TweenInfo.new(0.2), {Color = self.theme.Outline}):Play() end)

	local submitButton = self:Create("TextButton", {
		Name = "Submit",
		Position = UDim2.new(0.5, -75, 1, -55),
		Size = UDim2.new(0, 150, 0, 40),
		BackgroundColor3 = self.theme.Primary,
		Font = self.theme.Font.Bold,
		Text = "Submit",
		TextColor3 = self.theme.OnPrimary,
		TextSize = 16,
		Parent = main
	})
	self:Create("UICorner", { CornerRadius = UDim.new(0, self.theme.ElementRounding), Parent = submitButton })

	function dialog:Show()
		overlay.Visible = true
		main.Position = UDim2.new(0.5, 0, 0.45, 0)
		main.BackgroundTransparency = 1
		tweenService:Create(overlay, TweenInfo.new(0.3, Enum.EasingStyle.Quad), {BackgroundTransparency = 0.5}):Play()
		tweenService:Create(main, TweenInfo.new(0.4, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {Position = UDim2.new(0.5, 0, 0.5, 0), BackgroundTransparency = 0.1}):Play()
	end

	function dialog:Hide()
		local tween = tweenService:Create(overlay, TweenInfo.new(0.3, Enum.EasingStyle.Quad), {BackgroundTransparency = 1})
		tween:Play()
		tween.Completed:Connect(function() overlay:Destroy() end)
	end
	
	submitButton.MouseButton1Click:Connect(function()
		statusLabel.Text = "Checking..."
		statusLabel.TextColor3 = self.theme.Muted
		
		coroutine.wrap(function()
			local success, result = pcall(checkCallback, keyBox.Text)
			
			if success and result then
				statusLabel.Text = "Success! Welcome."
				statusLabel.TextColor3 = Color3.fromRGB(120, 255, 120)
				wait(1)
				dialog:Hide()
				successCallback()
			else
				statusLabel.Text = "Invalid key. Please try again."
				statusLabel.TextColor3 = Color3.fromRGB(255, 120, 120)
				local originalPos = main.Position
				for i = 1, 5 do
					main.Position = originalPos + UDim2.new(0, math.random(-5, 5), 0, 0)
					task.wait(0.05)
				end
				main.Position = originalPos
				
				-- If check fails multiple times, execute failure callback
				options.failCount = (options.failCount or 0) + 1
				if options.failCount >= 3 then
					task.wait(0.5)
					failureCallback()
				end
			end
		end)()
	end)

	return dialog
end

			for _, data in pairs(window.tabPages) do
				if data.page.Visible then
					totalContentHeight = data.layout.AbsoluteContentSize.Y
					break
				end
			end
		else
			totalContentHeight = mainLayout.AbsoluteContentSize.Y
		end
		
		local newHeight = contentY + totalContentHeight
		window.main:TweenSize(UDim2.new(0, containerWidth, 0, newHeight), "Out", "Quad", 0.2, true)
	end

	if window.tabs then
		for _, data in pairs(window.tabPages) do
			data.layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(updateSize)
		end
	else
		mainLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(updateSize)
	end
	
	titleFrame.InputBegan:connect(function(input) if input.UserInputType == ui then dragging, dragStart, startPos, dragObject = true, input.Position, window.main.Position, window.main end end)
	titleFrame.InputChanged:connect(function(input) if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then dragInput = input end end)
	titleFrame.InputEnded:connect(function(input) if input.UserInputType == ui then dragging = false end end)

	function window:SetTitle(newTitle)
		title.Text = tostring(newTitle)
	end

	return window
end

function createToggle(option, parent)
	local main = library:Create("TextButton", { LayoutOrder = option.position, Size = UDim2.new(1, 0, 0, 36), BackgroundTransparency = 1, Text = "", AutoButtonColor = false, Parent = parent })
	local label = library:Create("TextLabel", { Position = UDim2.new(0, 0, 0, 0), Size = UDim2.new(1, -50, 1, 0), BackgroundTransparency = 1, Text = option.text, TextSize = 16, Font = library.theme.Font.Regular, TextColor3 = library.theme.OnBackground, TextXAlignment = Enum.TextXAlignment.Left, Parent = main })
	if option.icon then createIcon(label, option.icon, library.theme.OnBackground); label.Text = "   " .. label.Text end
	local switchTrack = library:Create("Frame", { Name = "SwitchTrack", AnchorPoint = Vector2.new(1, 0.5), Position = UDim2.new(1, 0, 0.5, 0), Size = UDim2.new(0, 40, 0, 22), BackgroundColor3 = option.state and library.theme.Primary or library.theme.Surface, BorderSizePixel = 0, Parent = main })
	library:Create("UICorner", { CornerRadius = UDim.new(0, library.theme.SmallRounding), Parent = switchTrack })
	local switchThumb = library:Create("Frame", { Name = "SwitchThumb", AnchorPoint = Vector2.new(0.5, 0.5), Position = option.state and UDim2.new(1, -11, 0.5, 0) or UDim2.new(0, 11, 0.5, 0), Size = UDim2.new(0, 16, 0, 16), BackgroundColor3 = option.state and library.theme.OnPrimary or library.theme.OutlineVariant, BorderSizePixel = 0, Parent = switchTrack })
	library:Create("UICorner", { CornerRadius = UDim.new(0, library.theme.SmallRounding), Parent = switchThumb })
	function option:SetState(state)
		library.flags[self.flag] = state; self.state = state
		tweenService:Create(switchTrack, TweenInfo.new(0.2), {BackgroundColor3 = state and library.theme.Primary or library.theme.Surface}):Play()
		tweenService:Create(switchThumb, TweenInfo.new(0.2), { Position = state and UDim2.new(1, -11, 0.5, 0) or UDim2.new(0, 11, 0.5, 0), BackgroundColor3 = state and library.theme.OnPrimary or library.theme.OutlineVariant }):Play()
		self.callback(state)
	end
	main.MouseButton1Click:Connect(function() option:SetState(not option.state) end)
	if option.state then delay(0.1, function() option.callback(true) end) end
	setmetatable(option, {__newindex = function(t, i, v) if i == "Text" then label.Text = (option.icon and "   " or "") .. tostring(v) end end})
end

local function createBind(option, parent)
	local binding = false
	local text = string.match(option.key, "Mouse") and string.sub(option.key, 1, 5) .. string.sub(option.key, 12, 13) or option.key
	local main = library:Create("Frame", { LayoutOrder = option.position, Size = UDim2.new(1, 0, 0, 36), BackgroundTransparency = 1, Parent = parent })
	local label = library:Create("TextLabel", { Position = UDim2.new(0, 0, 0, 0), Size = UDim2.new(1, -80, 1, 0), BackgroundTransparency = 1, Text = option.text, TextSize = 16, Font = library.theme.Font.Regular, TextColor3 = library.theme.OnBackground, TextXAlignment = Enum.TextXAlignment.Left, Parent = main })
	if option.icon then createIcon(label, option.icon, library.theme.OnBackground); label.Text = "   " .. label.Text end
	local bindButton = library:Create("TextButton", { AnchorPoint = Vector2.new(1, 0.5), Position = UDim2.new(1, 0, 0.5, 0), Size = UDim2.new(0, 70, 1, -4), BackgroundColor3 = library.theme.Surface, Font = library.theme.Font.Semibold, Text = text, TextColor3 = library.theme.OnSurface, TextSize = 14, AutoButtonColor = false, Parent = main })
	library:Create("UICorner", { CornerRadius = UDim.new(0, library.theme.ElementRounding), Parent = bindButton })
	bindButton.MouseButton1Click:Connect(function() binding = true; bindButton.Text = "..."; tweenService:Create(bindButton, TweenInfo.new(0.2), {BackgroundColor3 = library.theme.Primary}):Play() end)
	local function processInput(input)
		if binding then
			local keyName = input.UserInputType.Name:find("Mouse") and input.UserInputType.Name or (input.KeyCode ~= Enum.KeyCode.Unknown and input.KeyCode.Name)
			if keyName then option:SetKey(keyName) end
		elseif (input.KeyCode.Name == option.key or input.UserInputType.Name == option.key) then option.callback() end
	end
	inputService.InputBegan:Connect(function(input) if not inputService:GetFocusedTextBox() then processInput(input) end end)
	function option:SetKey(key)
		binding = false; self.key = key; library.flags[self.flag] = self.key
		local displayText = string.match(self.key, "Mouse") and string.sub(self.key, 1, 5) .. string.sub(self.key, 12, 13) or self.key
		bindButton.Text = displayText
		tweenService:Create(bindButton, TweenInfo.new(0.2), {BackgroundColor3 = library.theme.Surface}):Play()
	end
end

local function createSlider(option, parent)
	local main = library:Create("Frame", { LayoutOrder = option.position, Size = UDim2.new(1, 0, 0, 50), BackgroundTransparency = 1, Parent = parent })
	local title = library:Create("TextLabel", { Position = UDim2.new(0, 0, 0, 0), Size = UDim2.new(1, -50, 0, 20), BackgroundTransparency = 1, Text = option.text, TextSize = 16, Font = library.theme.Font.Regular, TextColor3 = library.theme.OnBackground, TextXAlignment = Enum.TextXAlignment.Left, Parent = main })
	local valueLabel = library:Create("TextLabel", { Position = UDim2.new(1, 0, 0, 0), AnchorPoint = Vector2.new(1, 0), Size = UDim2.new(0, 50, 0, 20), BackgroundTransparency = 1, Text = tostring(option.value), TextSize = 15, Font = library.theme.Font.Semibold, TextColor3 = library.theme.Primary, TextXAlignment = Enum.TextXAlignment.Right, Parent = main })
	local sliderTrack = library:Create("Frame", { Position = UDim2.new(0.5, 0, 1, -14), AnchorPoint = Vector2.new(0.5, 1), Size = UDim2.new(1, 0, 0, 6), BackgroundColor3 = library.theme.Surface, BorderSizePixel = 0, Parent = main })
	library:Create("UICorner", { CornerRadius = UDim.new(0, library.theme.SmallRounding), Parent = sliderTrack })
	local fill = library:Create("Frame", { Size = UDim2.new((option.value - option.min) / (option.max - option.min), 0, 1, 0), BackgroundColor3 = library.theme.Primary, BorderSizePixel = 0, Parent = sliderTrack })
	library:Create("UICorner", { CornerRadius = UDim.new(0, library.theme.SmallRounding), Parent = fill })
	local thumb = library:Create("Frame", { AnchorPoint = Vector2.new(0.5, 0.5), Position = UDim2.new((option.value - option.min) / (option.max - option.min), 0, 0.5, 0), Size = UDim2.new(0, 16, 0, 16), BackgroundColor3 = library.theme.Primary, BorderSizePixel = 0, Parent = sliderTrack })
	library:Create("UICorner", { CornerRadius = UDim.new(0, library.theme.SmallRounding), Parent = thumb })
	local sliding = false
	function option:SetValue(value)
		value = round(value, option.float); value = math.clamp(value, self.min, self.max)
		local percentage = (value - self.min) / (self.max - self.min)
		thumb:TweenPosition(UDim2.new(percentage, 0, 0.5, 0), "Out", "Quad", 0.05, true)
		fill:TweenSize(UDim2.new(percentage, 0, 1, 0), "Out", "Quad", 0.05, true)
		library.flags[self.flag] = value; self.value = value; valueLabel.Text = tostring(value); self.callback(value)
	end
	local function updateFromInput(input) local percentage = (input.Position.X - sliderTrack.AbsolutePosition.X) / sliderTrack.AbsoluteSize.X; option:SetValue(option.min + percentage * (option.max - option.min)) end
	main.InputBegan:Connect(function(input) if input.UserInputType == ui then sliding = true; tweenService:Create(thumb, TweenInfo.new(0.2), {Size = UDim2.new(0, 20, 0, 20)}):Play(); updateFromInput(input) end end)
	inputService.InputChanged:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseMovement and sliding then updateFromInput(input) end end)
	inputService.InputEnded:Connect(function(input) if input.UserInputType == ui and sliding then sliding = false; tweenService:Create(thumb, TweenInfo.new(0.2), {Size = UDim2.new(0, 16, 0, 16)}):Play() end end)
end

local function loadOptions(options, parent)
	for _,newOption in next, options do
		if newOption.type == "label" then createLabel(newOption, parent)
		elseif newOption.type == "toggle" then createToggle(newOption, parent)
		elseif newOption.type == "button" then createButton(newOption, parent)
		elseif newOption.type == "bind" then createBind(newOption, parent)
		elseif newOption.type == "slider" then createSlider(newOption, parent)
		elseif newOption.type == "info" then createInfo(newOption, parent)
		elseif newOption.type == "separator" then createSeparator(newOption, parent)
		-- Other elements like List, Box, ColorPicker would be here
		end
	end
end

local function getFnctions(parent)
	local function getTargetContent()
		if parent.tabs and parent.activeTab then
			return parent.tabPages[parent.activeTab].page
		end
		return parent.content
	end
	
	function parent:AddLabel(option) option = type(option) == "table" and option or {text = option}; option.type = "label"; option.position = #self.options; table.insert(self.options, option); return option end
	function parent:AddToggle(option) option.type = "toggle"; option.position = #self.options; option.flag = option.flag or option.text; library.flags[option.flag] = option.state or false; table.insert(self.options, option); return option end
	function parent:AddButton(option) option.type = "button"; option.position = #self.options; option.flag = option.flag or option.text; table.insert(self.options, option); return option end
	function parent:AddBind(option) option.type = "bind"; option.position = #self.options; option.key = (option.key and option.key.Name) or option.key or "F"; option.flag = option.flag or option.text; library.flags[option.flag] = option.key; table.insert(self.options, option); return option end
	function parent:AddSlider(option) option.type = "slider"; option.position = #self.options; option.min = option.min or 0; option.max = option.max or 100; option.value = math.clamp(option.value or option.min, option.min, option.max); option.float = option.float or 1; option.flag = option.flag or option.text; library.flags[option.flag] = option.value; table.insert(self.options, option); return option end
	function parent:AddInfo(option) option = type(option) == "table" and option or {text = option}; option.type = "info"; option.position = #self.options; table.insert(self.options, option); return option end
	function parent:AddSeparator() local option = {type = "separator", position = #self.options}; table.insert(self.options, option); return option end
	
	-- Folder and other complex elements would be defined here
end

function library:CreateWindow(options)
	assert(type(options) == "table", "CreateWindow now expects a table of options.")
	local window = {
		title = options.Title or "Window",
		tag = options.Tag,
		tabs = options.Tabs,
		options = {},
		open = true,
		canInit = true,
		init = false,
		position = #self.windows
	}
	
	if window.tabs then
		window.tabOptions = {}
		for _, tabName in ipairs(window.tabs) do
			window.tabOptions[tabName] = {}
		end
		
		function window:Tab(tabName)
			assert(window.tabOptions[tabName], "Invalid tab name: " .. tostring(tabName))
			local tabInterface = {}
			getFnctions(tabInterface)
			-- Override the options table to point to the correct tab's options
			tabInterface.options = window.tabOptions[tabName]
			return tabInterface
		end
	else
		getFnctions(window)
	end
	
	table.insert(library.windows, window)
	return window
end

function library:Init()
	self.base = self.base or self:Create("ScreenGui")
	if syn and syn.protect_gui then syn.protect_gui(self.base)
	elseif get_hidden_gui then self.base.Parent = get_hidden_gui()
	elseif gethui then self.base.Parent = gethui()
	else self.base.Parent = game:GetService"CoreGui" end
	self.base.ResetOnSpawn = false
	self.base.Name = "XyreonLib_" .. tostring(math.random(1, 1000))
	
	for _, window in next, self.windows do
		if window.canInit and not window.init then
			window.init = true
			createOptionHolder(window, self.base)
			if window.tabs then
				for tabName, options in pairs(window.tabOptions) do
					loadOptions(options, window.tabPages[tabName].page)
				end
			else
				loadOptions(window.options, window.content)
			end
		end
	end
	return self.base
end

function library:Close()
	if typeof(self.base) ~= "Instance" then return end
	self.open = not self.open
	if self.activePopup then self.activePopup:Close() end
	for _, window in next, self.windows do
		if window.main then window.main.Visible = self.open end
	end
end

inputService.InputBegan:connect(function(input)
	if input.UserInputType == ui and library.activePopup then
		local pos, popup = input.Position, library.activePopup.mainHolder
		local absPos, absSize = popup.AbsolutePosition, popup.AbsoluteSize
		if not (pos.X >= absPos.X and pos.X <= absPos.X + absSize.X and pos.Y >= absPos.Y and pos.Y <= absPos.Y + absSize.Y) then
			library.activePopup:Close()
		end
	end
end)

inputService.InputChanged:connect(function(input)
	if input == dragInput and dragging then update(input) end
end)

return library
